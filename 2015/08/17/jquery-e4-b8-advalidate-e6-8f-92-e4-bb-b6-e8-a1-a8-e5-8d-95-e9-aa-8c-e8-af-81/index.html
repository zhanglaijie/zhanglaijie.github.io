        var validator;

            $.validator.addMethod(“zip_code“,function(value,element,parents){
                var laijie =/^[0-9]{6}$/;
                return this.optional(element)||(laijie.test(value));
            },$.validator.format(“{0}“));

            $(“#check“).click(function(){
                alert($(“#validateform“).valid()?“true“:“false“);

            validator = $(“#validateform“).validate({
                rules:{
                    username:{
                        required:true,
                        rangelength:[2,10],

                    },
                    passwd:{
                        required:true,
                        rangelength:[2,16]
                    },
                    confirmpasswd:{
                        required:true,
                        equalTo:“#passwd“
                    }
                },
                messages:{
                    username:{
                        required:“户名必填“,
                        rangelength:“用户名介于{0}和{1}之间“
                    },
                    passwd:{
                        required:“密码为空“,
                        rangelength:“密码介于{0}和{1}之间“
                    },
                    confirmpasswd:{
                        required:“确认密码为空“,
                        equalTo:“密码不相同“
                    }
                },
                submitHandler:function(form){
                    //form.submit();
                    console.log($(form).serialize());

            });
        });
